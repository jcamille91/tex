	\begin{frame}
		\frametitle{neutrinos in the Standard Model}
		\begin{columns}[c] % The "c" option specifies centered vertical alignment while the "t" option is used for top vertical alignment
			
			\column{.45\textwidth} % Left column and width
			%\textbf{Heading}
			\begin{itemize}
				\setlength\itemsep{2em}
				\item standard model leptons $\textit{e}, \mu,\tau$ each have associated neutrinos $\nu_e, \nu_{\mu}, \nu_{\tau}$. 
				\item Neutrinos only exist as left handed particles, so via the Higgs Mechanism and the Yuakawa interaction term, the Standard model predicts they are \footnotesize{\textbf{massless}}. \\ $\mathcal{L}_{mass} = m_{\ell} \bar{\ell_L} \ell_R$
			\end{itemize}
			
			\column{.5\textwidth} % Right column and width
			\footnotesize{U(1) gauge invariance}
			\begin{gather*}
			\begin{pmatrix}
			\nu_{\alpha} \\
			\ell_{\alpha}
			\end{pmatrix}
			\rightarrow e^{i\theta_{\alpha}}
			\begin{pmatrix}
			\nu_{\alpha} \\
			\ell_{\alpha}
			\end{pmatrix}
			\end{gather*}
			$\implies$ \footnotesize{\textbf{lepton flavor conservation}} \\
			$\implies$ $\nu_{\alpha} \nleftrightarrow \nu_{\beta}$
			
			
		\end{columns}
	\end{frame}
	
	\begin{frame}
		\frametitle{neutrino mixing}
			A given flavor eigenstate is a linear combination of mass eigenstates (PMNS-matrix)
			\begin{equation*}
			|\nu_{\alpha}\rangle = \sum_{i=1}^{3} U_{\alpha i} |\nu_{i}\rangle
			\end{equation*}
			parameters: angles $\theta_{12},\theta_{13},\theta_{23}$ CP-violating phases $\delta_{CP}, \alpha_1, \alpha_2$, and neutrino masses $m_1, m_2, m_3$
			\begin{eqnarray*}			
			& U = \begin{pmatrix}
			c_{12}c_{13} & s_{12}c_{13} & s_{13}e^{-i\delta_{CP}} \\
			-s_{12}c_{23}-c_{12}s_{23}s_{13}e^{i\delta_{CP}} & c_{12}c_{23}-s_{12}s_{23}s_{13}e^{i\delta_{CP}}
			& s_{23}c_{13} \\ s_{12}s_{23}-c_{12}c_{23}s_{13}e^{i\delta_{CP}} & 
			-c_{12}s_{23}-s_{12}c_{23}s_{13}e^{i\delta_{CP}} & c_{23c_{13}}
			\end{pmatrix} \\
			& \times 
			\begin{pmatrix}
			e^{i\alpha_1 /2} & 0 & 0 \\
			0 & e^{i\alpha_2 /2} & 0 \\
			0 & 0 & 1
			\end{pmatrix}
			\end{eqnarray*}
	\end{frame}
